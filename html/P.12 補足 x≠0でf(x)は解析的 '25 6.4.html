P.12 補足 x≠0でf(x)は解析的 '25 6.4

\[\begin{flalign*} f(x)= \begin{cases} e^{-\frac{1}{x^2}} & x\ne 0 \\ 0 & x=0 \end{cases} && \end{flalign*}\]

\(x\ne0\) で \(f(x)\) は解析的

(証明)

\(x\) は \(a\) を中心とするべき級数で表される \(\textcolor{red}{(*1)}\)

\[\begin{flalign*} &x=\displaystyle\sum_{n=0}^{\infty}a_n(x-a)^n, \quad a_n=\begin{cases} a & (n=0) \\ 1 & (n=1) \\ 0 & (n>1) \end{cases} & \end{flalign*}\]

収束半径は \(\infty\)

\[\begin{flalign*} &&\textcolor{red}{(*1)}&F(x)=\displaystyle\sum_{n=0}^{\infty}a_n(x-a)^n, a_n=\begin{cases}a & (n=0)\\ 1 & (n=1)\\ 0 & (n>1)\end{cases} とする\\ &&& F(x)=a_0(x-a)^0+a_1(x-a)^1+a_2(x-a)^2+\dots\\ &&& \qquad=a+(x-a)+0\\ &&& \qquad=x\\ &&& 任意のxで収束するので、収束半径は \infty \end{flalign*}\]

\(\displaystyle\frac{1}{x}\) は \(a\ne0\) を中心とするべき級数で表される \(\textcolor{red}{(*2)}\)

\[\begin{flalign*} &\frac{1}{x}=\displaystyle\sum_{n=0}^{\infty}b_n(x-a)^n, \quad b_n=(-1)^n \frac{1}{a^{n+1}}& \end{flalign*}\]

収束半径は \(|a|\)

\[\begin{flalign*} &&\textcolor{red}{(*2)}&a\ne0 とする \\ &&& G(x)=\displaystyle\sum_{n=0}^{\infty}b_n(x-a)^n とする。収束すると仮定する\\ &&& x = \displaystyle\sum_{n=0}^{\infty}a_n(x-a)^n, a_n=\begin{cases}a & (n=0)\\ 1 & (n=1)\\ 0 & (n>1)\end{cases} とする\\ &&& 1 = x G(x) とする\\ &&& 1 = \displaystyle\sum_{n=0}^{\infty}a_n(x-a)^n\displaystyle\sum_{n=0}^{\infty}b_n(x-a)^n \\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}\big(\displaystyle\sum_{k=0}^{n}a_k b_{n-k}\big)(x-a)^n ({\because}コーシー積より)\\ &&& \quad= \displaystyle\sum_{k=0}^{0}a_k b_{0-k}(x-a)^0 +\displaystyle\sum_{k=0}^{1}a_k b_{1-k}(x-a)^1 +\displaystyle\sum_{k=0}^{2}a_k b_{2-k}(x-a)^2 +\dots\\ &&& 1=\displaystyle\sum_{k=0}^{0}a_k b_{0-k} と仮定する\ 1=a_0 b_0\ {\therefore}b_0=\frac{1}{a_0}=\frac{1}{a} \\ &&& 0=\displaystyle\sum_{k=0}^{n}a_k b_{n-k}\ (n>0)と仮定する\\ &&& 0=a_0 b_0+a_1 b_{n-1}+\displaystyle\sum_{k=2}^{n}a_k b_{n-k}\\ &&& {\therefore}0=a b_n + b_{n-1} + 0\ ({\because}a_0=a,a_1=1,a_k=0(k>1))\\ &&& {\therefore}b_n=-\frac{1}{a}b_{n-1}\\ &&& {\therefore}b_n=\frac{1}{a}\Big(-\frac{1}{a}\Big)^n=(-1)^n\frac{1}{a^{n+1}}\\ &&& これを踏まえて \\ &&& a\ne0 とする \\ &&& G(x)=\displaystyle\sum_{n=0}^{\infty}b_n(x-a)^n, b_n=(-1)^n\frac{1}{a^{n+1}}とする\\ &&& G(x) は初項 \frac{1}{a}、 公比 \frac{-1}{a}(x-a)の等比級数\\ &&& よって\Big| \frac{-1}{a}(x-a) \Big|<1 即ち |x-a|<|a| ならば収束する\\ &&& よって |x-a|<|a| ならば\\ &&& G(x)=\frac{1}{a}\frac{1}{1-\frac{-1}{a}(x-a)}=\frac{1}{a+(x-a)}=\frac{1}{x}\ ({\because}等比級数の公式)\\ &&& また 等比級数なので \Big| \frac{-1}{a}(x-a) \Big|>1 即ち |x-a|>|a| ならば収束しない\\ &&& よって 収束半径は |a| \end{flalign*}\]

\(\displaystyle\frac{1}{x^2}\) は \(a\ne0\) を中心とするべき級数で表される \(\textcolor{red}{(*3)}\)

\(|x-a|<|a|\) ならば

\[\begin{flalign*} &\frac{1}{x^2}=\displaystyle\sum_{n=0}^{\infty}c_n(x-a)^n, \quad c_n=(-1)^n \frac{n+1}{a^{n+2}}& \end{flalign*}\]

級数は絶対収束する。

\[\begin{flalign*} &&\textcolor{red}{(*3)}&|x-a|<|a| とする \\ &&& \frac{1}{x}=\displaystyle\sum_{n=0}^{\infty}b_n(x-a)^n,b_n=(-1)^n\frac{1}{a^{n+1}} とする\\ &&& 級数は絶対収束する。よって\\ &&& \frac{1}{x^2}=\frac{1}{x}\frac{1}{x} =\displaystyle\sum_{n=0}^{\infty}b_n(x-a)^n \displaystyle\sum_{n=0}^{\infty}b_n(x-a)^n \\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}\displaystyle\sum_{k=0}^{n}b_k(x-a)^k b_{n-k}(x-a)^{n-k}\dots(1) \left(\begin{array}{@{}l@{}}{\because}絶対収束する級数の積は \\ \quad コーシー積に等しい \\ \quad またコーシー積は絶対収束する \end{array} \right)\\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}\Big(\displaystyle\sum_{k=0}^{n}b_k b_{n-k} \Big)(x-a)^n \ ({\because}有限級数の線型性)\\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}\Big(\displaystyle\sum_{k=0}^{n}(-1)^k\frac{1}{a^{k+1}}(-1)^{n-k}\frac{1}{a^{n-k+1}}\Big)(x-a)^n\\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}\Big(\displaystyle\sum_{k=0}^{n}(-1)^n\frac{1}{a^{n+1}}\Big)(x-a)^n\\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}\Big((-1)^n\frac{1}{a^{n+1}}\displaystyle\sum_{k=0}^{n}1\Big)(x-a)^n\\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}(-1)^n\frac{1}{a^{n+2}}(n+1)(x-a)^n\\ &&& 絶対収束する級数の積をあらわすコーシー積は絶対収束する\\ &&& よって(1)よりこの級数は絶対収束する\\ &&& (もしくは、収束するべき級数は絶対収束するので この級数は絶対収束する) \end{flalign*}\]

\(\displaystyle-\frac{1}{x^2}\) は \(a\ne0\) を中心とするべき級数で表される \(\textcolor{red}{(*4)}\)

\(|x-a|<|a|\) ならば

\[\begin{flalign*} &-\frac{1}{x^2}=\displaystyle\sum_{n=0}^{\infty}s_n(x-a)^n, \quad s_n=(-1)^{n+1}\frac{n+1}{a^{n+2}}& \end{flalign*}\]

級数は絶対収束する。

\[\begin{flalign*} &&\textcolor{red}{(*4)}&|x-a|<|a| とする \\ &&& \frac{1}{x^2}=\displaystyle\sum_{n=0}^{\infty}(-1)^n \frac{n+1}{a^{n+2}}(x-a)^n\\ &&& よって\\ &&& -\frac{1}{x^2}=(-1)\displaystyle\sum_{n=0}^{\infty}(-1)^n \frac{n+1}{a^{n+2}}(x-a)^n\\ &&& \quad=\displaystyle\sum_{n=0}^{\infty}(-1)^{n+1} \frac{n+1}{a^{n+2}}(x-a)^n \ ({\because}絶対収束する級数は線型性をもつ)\\ &&& また\displaystyle\sum_{n=0}^{\infty}\Big|(-1)^{n+1}\frac{n+1}{a^{n+2}}(x-a)^n\Big| =\displaystyle\sum_{n=0}^{\infty}\Big|(-1)^{n}\frac{n+1}{a^{n+2}}(x-a)^n\Big|\\ &&& \frac{1}{x^2}の級数が絶対収束するので右辺は収束する\\ &&& よって-\frac{1}{x^2}の級数は絶対収束する \end{flalign*}\]

\(e^x\) は \(0\) を中心とするべき級数で表される

\[\begin{flalign*} &e^x = \displaystyle\sum_{n=0}^{\infty}\frac{x^n}{n!}\quad({\because}e^x の定義)& \end{flalign*}\]

すべての \(x\) について収束する \(\textcolor{red}{(*5)}\) よって収束半径は \(\infty\)

\[\begin{flalign*} &&\textcolor{red}{(*5)}& \sum \Big|\frac{x^n}{n!}\Big| について\\ &&& \displaystyle\lim_{n\to \infty} \frac{ \displaystyle\Big|\frac{x^{n+1}}{(n+1)!}\Big| }{\displaystyle\Big|\frac{x^n}{n!}\Big|} = \displaystyle\lim_{n\to \infty}\Big|\frac{x}{n+1}\Big|=0 \\ &&& よってダランベールの判定法より \sum \Big|\frac{x^n}{n!}\Big| は収束する \\ &&& よって \sum \frac{x^n}{n!} は収束する \end{flalign*}\]



最後に \(\displaystyle e^{-\frac{1}{x^2}}\) のべき級数を求める。

\(\displaystyle e^x = \displaystyle\sum_{n=0}^{\infty}a_n x^n,\ a_n=\frac{1}{n!}\)とする

\(a\ne0,\ |x-a|<|a|\) とする

\(\displaystyle-\frac{1}{x^2}=\displaystyle\sum_{m=0}^{\infty} s_m(x-a)^n, \ s_m=(-1)^{m+1}\frac{m+1}{a^{m+2}}\) とする

べき級数の合成(別頁) より

\(\displaystyle\displaystyle\sum_{m=0}^{\infty}|s_m(x-a)^m|<\infty\) ならば

\(\displaystyle e^{-\frac{1}{x^2}} = \displaystyle\sum_{p=0}^{\infty}d_p(x-p)^p\) \(,\quad d_p = \displaystyle\displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \displaystyle\sum_{k_1+\dots+k_n=p}^{}s_{k_1}\dots s_{k_n}\)

である

\[\begin{flalign*} d_p &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \displaystyle\sum_{k_1+\dots+k_n=p}^{} (-1)^{k_1+1}\frac{k_1+1}{a^{k_1+2}} \dots (-1)^{k_n+1}\frac{k_n+1}{a^{k_n+2}} &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \displaystyle\sum_{k_1+\dots+k_n=p}^{} (-1)^{p+n}\frac{(k_1+1)\dots(k_n+1)}{a^{p+2n}} &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^p \Big(\frac{-1}{a^2}\Big)^n \displaystyle\sum_{k_1+\dots+k_n=p}^{}(k_1+1)\dots(k_n+1) \ ({\because}有限級数の線型性) & \end{flalign*}\]

\(a\ne0,\ |x-a|<|a|\) ならば \(\displaystyle-\frac{1}{x^2}=\displaystyle\sum_{m=0}^{\infty} s_m(x-a)^n\) は絶対収束する

よって \(\displaystyle\displaystyle\sum_{m=0}^{\infty} |s_m(x-a)^n|\) は収束する

よって \(\displaystyle\displaystyle\sum_{m=0}^{\infty} |s_m(x-a)^n| < \infty\)

よって \(a\ne0,\ |x-a|<|a|\) ならば

\[\begin{flalign*} e^{-\frac{1}{x^2}} &= \displaystyle\sum_{p=0}^{\infty}d_p(x-a)^p &\\ d_p &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^p \Big(\frac{-1}{a^2}\Big)^n \displaystyle\sum_{k_1+\dots+k_n=p}^{}(k_1+1)\dots(k_n+1) & \end{flalign*}\]

\(\displaystyle e^{-\frac{1}{x^2}}\) は \(a\ne0\) を中心とするべき級数であらわされる。よって解析的である。

\(x\ne0\) で \(f(x)=\displaystyle e^{-\frac{1}{x^2}}\) なので \(x\ne0\) で \(f(x)\) は解析的である。

(収束性について)

\(a\ne0,\ |x-a|<|a|\) において \(-\frac{1}{x^2}\) と \(e^x\) の級数は絶対収束するので、コーシー積の \(e^{-\frac{1}{x^2}}\) の級数も絶対収束する

(注)「収束半径＝一番近い特異点までの距離」は実関数では成立しないので簡単に収束半径 \(|a|\) とは言えない

(最初の3項を求めてみる)

\[\begin{flalign*} d_0 &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^0 \Big(\frac{-1}{a^2}\Big)^n \displaystyle\sum_{k_1+\dots+k_n=0}^{}(k_1+1)\dots(k_n+1) &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a^2}\Big)^n \cdot 1 &\\ &= e^{-\frac{1}{a^2}} &\\ d_1 &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^1 \Big(\frac{-1}{a^2}\Big)^n \displaystyle\sum_{k_1+\dots+k_n=0}^{}(k_1+1)\dots(k_n+1) &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big) \Big(\frac{-1}{a^2}\Big)^n {}_n C_1 &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{(n-1)!} \Big(\frac{-1}{a}\Big) \Big(\frac{-1}{a^2}\Big) \Big(\frac{-1}{a^2}\Big)^{n-1} &\\ &= \frac{1}{a^3} \displaystyle\sum_{n=0}^{\infty}\frac{1}{(n-1)!} \Big(\frac{-1}{a^2}\Big)^{n-1} \ \left(\begin{array}{@{}l@{}} {\because}(別頁):べき級数の合成より dp は絶対収束する \\ \quad なので線型性をもつ \\ \end{array} \right) &\\ &= \frac{1}{a^3}e^{-\frac{1}{a^2}} &\\ d_2 &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^2 \Big(\frac{-1}{a^2}\Big)^n \displaystyle\sum_{k_1+\dots+k_n=2}^{}(k_1+1)\dots(k_n+1) &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^2 \Big(\frac{-1}{a^2}\Big)^n (4{}_n C_2+3{}_n C_1) &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^2 \Big(\frac{-1}{a^2}\Big)^n (2n(n-1)+3n) &\\ &= \displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^2 \Big(\frac{-1}{a^2}\Big)^n 2n(n-1) +\displaystyle\sum_{n=0}^{\infty}\frac{1}{n!} \Big(\frac{-1}{a}\Big)^2 \Big(\frac{-1}{a^2}\Big)^n 3n\ ({\because}dp の線型性) &\\ &= \frac{2}{a^6} \displaystyle\sum_{n=0}^{\infty}\frac{1}{(n-2)!} \Big(\frac{-1}{a^2}\Big)^{n-2} +\frac{-3}{a^4} \displaystyle\sum_{n=0}^{\infty}\frac{1}{(n-1)!} \Big(\frac{-1}{a^2}\Big)^{n-1}\ ({\because}dp の線型性) &\\ &= \frac{2}{a^6}e^{-\frac{1}{a^2}} +\frac{-3}{a^4}e^{-\frac{1}{a^2}}\ ({\because}e^x の定義) &\\ &= \Big(\frac{2}{a^6} -\frac{3}{a^4}\Big)e^{-\frac{1}{a^2}} & \end{flalign*}\]

よって

\[\begin{flalign*} e^{-\frac{1}{x^2}} &\approx e^{-\frac{1}{a^2}} +\frac{1}{a^2}e^{-\frac{1}{a^2}} +\frac{1}{a^3}e^{-\frac{1}{a^2}} +\Big(\frac{2}{a^6} -\frac{3}{a^4}\Big)e^{-\frac{1}{a^2}} & \end{flalign*}\]
