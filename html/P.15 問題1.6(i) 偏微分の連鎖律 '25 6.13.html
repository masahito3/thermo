\(\pmb{\textcolor{blue}{Z}}(x,y)\), \(\pmb{\textcolor{magenta}{Z}}(\xi,\eta)\) とする

\(\displaystyle\Big(\frac{\partial \pmb{\textcolor{magenta}{Z}}}{\partial \xi}\Big)_\eta= \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial x}\Big)_y \displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{x}}}{\partial \xi}\Big)_\eta + \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial y}\Big)_x \displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{y}}}{\partial \xi}\Big)_\eta\) \(\cdots(1.20)\)

(証明)

\(x,y,\xi,\eta\) は独立変数とする

\(\pmb{\textcolor{blue}{Z}}, \pmb{\textcolor{magenta}{Z}}, \pmb{\textcolor{teal}{x}}, \pmb{\textcolor{teal}{y}}\) は関数とする

\(\pmb{\textcolor{blue}{Z}}(x,y)\) とする

\(\pmb{\textcolor{teal}{x}}(\xi,\eta)\) とする

\(\pmb{\textcolor{teal}{y}}(\xi,\eta)\) とする

\(\pmb{\textcolor{magenta}{Z}}(\xi,\eta)=\pmb{\textcolor{blue}{Z}}(\pmb{\textcolor{teal}{x}}(\xi,\eta),\pmb{\textcolor{teal}{y}}(\xi,\eta))\) とする

\[ \begin{flalign*} {\therefore}\displaystyle\Big(\frac{\partial \pmb{\textcolor{magenta}{Z}}}{\partial \xi}\Big)_\eta &= \frac{d\pmb{\textcolor{magenta}{Z}}}{d\xi} \quad \Big({\because}\xi,\eta が独立なので \displaystyle\Big(\frac{\partial \pmb{\textcolor{magenta}{Z}}}{\partial \xi}\Big)_\eta=\frac{d\pmb{\textcolor{magenta}{Z}}}{d\xi}\Big) &\\ &= \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial x}\Big)_y\frac{d\pmb{\textcolor{teal}{x}}}{d\xi}+\displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial y}\Big)_x\frac{d\pmb{\textcolor{teal}{y}}}{d\xi} \quad ({\because}問題1.7) &\\ &= \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial x}\Big)_y\displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{x}}}{\partial \xi}\Big)_\eta + \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial y}\Big)_x\displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{y}}}{\partial \xi}\Big)_\eta \quad \Big({\because}\xi,\eta が独立なので \displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{x}}}{\partial \xi}\Big)_\eta=\frac{d\pmb{\textcolor{teal}{x}}}{d\xi},\ \displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{y}}}{\partial \xi}\Big)_\eta=\frac{d\pmb{\textcolor{teal}{y}}}{d\xi}\Big) & \end{flalign*} \]
\[ \begin{flalign*} {\therefore}\displaystyle\Big(\frac{\partial \pmb{\textcolor{magenta}{Z}}}{\partial \eta}\Big)_\xi &= \frac{d\pmb{\textcolor{magenta}{Z}}}{d\eta} \quad \Big({\because}\xi,\eta が独立なので \displaystyle\Big(\frac{\partial \pmb{\textcolor{magenta}{Z}}}{\partial \eta}\Big)_\xi=\frac{d\pmb{\textcolor{magenta}{Z}}}{d\eta}\Big) &\\ &= \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial x}\Big)_y\frac{d\pmb{\textcolor{teal}{x}}}{d\eta}+\displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial y}\Big)_x\frac{d\pmb{\textcolor{teal}{y}}}{d\eta} \quad ({\because}問題1.7) &\\ &= \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial x}\Big)_y\displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{x}}}{\partial \eta}\Big)_\xi + \displaystyle\Big(\frac{\partial \pmb{\textcolor{blue}{Z}}}{\partial y}\Big)_x\displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{y}}}{\partial \eta}\Big)_\xi \quad \Big({\because}\xi,\eta が独立なので \displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{x}}}{\partial \eta}\Big)_\xi=\frac{d\pmb{\textcolor{teal}{x}}}{d\eta},\ \displaystyle\Big(\frac{\partial \pmb{\textcolor{teal}{y}}}{\partial \eta}\Big)_\xi=\frac{d\pmb{\textcolor{teal}{y}}}{d\eta}\Big) & \end{flalign*} \]
