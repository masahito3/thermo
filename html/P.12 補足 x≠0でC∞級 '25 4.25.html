P.12 補足 x≠0でC∞級 '25 4.25

\[\begin{flalign*} f(x)= \begin{cases} e^{-\frac{1}{x^2}} & x\ne 0 \\ 0 & x=0 \end{cases} && \end{flalign*}\] \(x\ne 0 で C^\infty 級\)

(証明)

\(x\ne 0とする\) \[\begin{flalign*} f^{(1)}&=\left(e^{-\frac{1}{x^2}}\right)' &\\ &= \left(-\frac{1}{x^2}\right)'e^{-\frac{1}{x^2}} \quad \textcolor{red}{(*1),(*2)}& \\ &= -\left(\frac{1}{x^2}\right)'e^{-\frac{1}{x^2}} \quad ({\because\ }積の微分)& \\ &= -(-2)x^{-3}e^{-\frac{1}{x^2}} \quad \textcolor{red}{(*3)}& \\ &= 2x^{-3}e^{-\frac{1}{x^2}} \quad\cdots(1)& \end{flalign*}\] である。

\(n>0 で\) \[\begin{flalign*} f^{(n)}&=\left(\sum_{\nu=1}^{m}k_\nu x^{-\nu}\right)e^{-\frac{1}{x^2}}& \end{flalign*}\]と仮定する

\[\begin{flalign*} \left(\sum_{\nu=1}^{m}k_\nu x^{-\nu}\right)'&=\sum_{\nu=1}^{m}k_\nu(x^{-\nu})'\quad({\because\ }和,積の微分)&\\ &=\sum_{\nu=1}^{m}(-\nu k_\nu)x^{-\nu-1}\quad\textcolor{red}{(*3)}\cdots(2)& \end{flalign*}\]なので

\[\begin{flalign*} f^{(n+1)}&=\left(\sum_{\nu=1}^{m}k_\nu x^{-\nu}\right)'e^{-\frac{1}{x^2}}+\left(\sum_{\nu=1}^{m}k_\nu x^{-\nu}\right)\left(e^{-\frac{1}{x^2}}\right)'\quad({\because\ }積の微分)&\\ &=\sum_{\nu=1}^{m}(-\nu k_{\nu})x^{-\nu-1}e^{-\frac{1}{x^2}}+\sum_{\nu=1}^{m}k_\nu x^{-\nu}2x^{-3}e^{-\frac{1}{x^2}}\quad({\because\ }(1),(2))&\\ &=\left(\sum_{\nu=1}^{m}-\nu k_\nu x^{-\nu-1}+\sum_{\nu=1}^{m}2k_{\nu}x^{-\nu-3}\right)e^{-\frac{1}{x^2}}&\\ &=\left(\sum_{i=2}^{m+1}-(i-1)k_{i-1}x^{-i}+\sum_{i=4}^{m+3}2k_{i-3}x^{-i}\right)e^{-\frac{1}{x^2}}&\\ &=\left((-1)k_1x^{-2}+(-2)k_2x^{-3} +\sum_{i=4}^{m+1}-(i-1)k_{i-1}x^{-i}\right.&\\ &\qquad\left.+\sum_{i=4}^{m+1}2k_{i-3}x^{-i} +2k_{m-1}x^{-(m+1)}+2k_mx^{-(m+3)}\right)e^{-\frac{1}{x^2}}&\\ &=\left((-1)k_1x^{-2}+(-2)k_2x^{-3} +\sum_{i=4}^{m+1}(-(i-1)k_{i-1}+2k_{i-3})x^{-i}\right.&\\ &\qquad\left.+2k_{m-1}x^{-(m+1)}+2k_mx^{-(m+3)}\right)e^{-\frac{1}{x^2}}& \end{flalign*}\]

ここで \[\begin{flalign*} p_i&= \begin{cases} 0 &(i=1)\\ -(i-1)k_{i-1} &(i=2,3) \\ -(i-1)k_{i-1}+2k_{i-3} &(i=4,\dots,m+1)\\ 2k_{i-3} &(i=m+2,m+3) \end{cases}&\\ s&=m+3& \end{flalign*}\] とする

\[\begin{flalign*} f^{(n+1)}&=\left(\sum_{i=1}^{s}p_ix^{-i}\right)e^{-1/x^2}& \end{flalign*}\]

よって、\(x\ne0,n>0\)において \[\begin{flalign*} f^{(n)}&=\left(\sum_{\nu=1}^{m}k_\nu x^{-\nu}\right)e^{-\frac{1}{x^2}}& \end{flalign*}\]である。

すべての\(n\)で\(f^{(n)}\)は存在するので\(f\)は\(C^\infty\)級である

\[\begin{flalign*} &&\textcolor{red}{(*1)}& 合成関数の微分 \\ &&&g'(x),f'(g(x)) が存在するなら \\ &&&f(g(x))'=g'(x)f'(g(x))\\ &&\textcolor{red}{(*2)}& (e^x)'=e^x \\ &&&(証明)\\ &&&e^x=\sum_{n=0}^{\infty}\frac{x^n}{n!}\ ({\because\ }e^x の定義)\\ &&&右辺の項別微分を考える\\ &&&\sum_{n=0}^{\infty}\left(\frac{x^n}{n!}\right)'=(1)'+\sum_{n=1}^{\infty}\left(\frac{x^n}{n!}\right)'\\ &&&\quad=\sum_{n=1}^{\infty}n\frac{x^{n-1}}{n!}\ \textcolor{red}{(*2.1)}\\ &&&\quad=\sum_{n=1}^{\infty}\frac{x^{n-1}}{(n-1)!}\\ &&&\quad=\sum_{n=0}^{\infty}\frac{x^n}{n!}\ \textcolor{red}{(*2.2)}\\ &&&\quad=e^x\ ({\because\ }e^x の定義)\\ &&&ここで任意のxに対して\\ &&&-A\le x\le A,A>0 なる区間を考える\\ &&&\left|\frac{x^\nu}{\nu!}\right|\le\frac{A^\nu}{\nu!},\nu=0,1,2,\ldots である\\ &&&また \sum_{\nu=0}^{\infty}\frac{A^\nu}{\nu!}=e^a\ ({\because\ }e^a の定義)\\ &&&なので \sum_{\nu=0}^{\infty}\frac{x^\nu}{\nu!} は区間[-A,A] で一様収束する\\ &&&({\because\ }定理:ある区間で |a_n(x)|\le C_n なる定数 C_n があって\\ &&&\quad \sum_{}^{\infty}C_n が収束するならば \sum_{}^{\infty}a_n は一様収束する)\\ &&&よって\left(e^x\right)'=e^x\\ &&&({\because\ }定理:無限級数が収束し各項の導関数が連続で項別微分が\\ &&&\quad 一様収束するならば無限級数の導関数は項別微分に等しい)\\ &&\textcolor{red}{(*2.1)}& (1)'=0\\ &&&n>0 ならば x^n=nx^{n-1}\ \textcolor{red}{(*3)}\\ &&&(kf(x))'=kf'(x)\ ({\because\ }積の微分)\\ &&\textcolor{red}{(*2.2)}&\sum_{n=1}^{\infty}\frac{x^{n-1}}{(n-1)!}=1+x+\frac{x^2}{2!}+\cdots\\ &&&\sum_{n=0}^{\infty}\frac{x^n}{n!}=1+x+\frac{x^2}{2!}+\cdots\\ &&&{\therefore\ }\sum_{n=1}^{\infty}\frac{x^{n-1}}{(n-1)!}=\sum_{n=0}^{\infty}\frac{x^n}{n!}\\ &&\textcolor{red}{(*3)}& x\ne 0,n>0 ならば \\ &&&(x^{-n})'=-nx^{-n-1}\\ &&&(証明)\\ &&&(x^n)'=-n x^{-n-1} と仮定する\\ &&&(x^{n+1})'=(xx^n)'=(x)'x^n+x(x^n)'\ ({\because\ }積の微分)\\ &&&\quad=x^n+xnx^{n-1}=(n+1)x^n\\ &&&(x)'=1=1\cdot x^{1-1}\\ &&&よって(x^n)'=nx^{n-1},n=1,2,\ldots\\ &&&よって x\ne 0, n=1,2,\ldots ならば\\ &&& (x^{-n})'=\left(\frac{1}{x^n}\right)'=\frac{1'x^n-1(x^n)'}{x^{2n}}\ ({\because\ }商の微分)\\ &&& \quad=\frac{-nx^{n-1}}{x^{2n}}=-nx^{-n-1} \end{flalign*}\]
