ゆっくり熱力学の基礎していってね

P.63 (4.10),(4.11) '25 8.5

\(\displaystyle S(U,V,N:C_1,C_2) =K\Big(\frac{UN}{2}\Big)^{1/3} \Bigg[\Bigg(\frac{V^{(1)}}{1+\sqrt{\frac{V-V^{(1)}}{V^{(1)}}}}\Bigg)^{1/3} +\Bigg(\frac{V-V^{(1)}}{1+\sqrt{\frac{V^{(1)}}{V-V^{(1)}}}}\Bigg)^{1/3}\Bigg]\)  (4.10)

\(\displaystyle U^{(1)}=\frac{U}{1+\sqrt{\frac{V-V^{(1)}}{V^{(1)}}}}\)  (4.11)

(証明)

\[ \begin{flalign*} \hat{S}(U^{(1)},V^{(1)},N^{(1)},U^{(2)},V^{(2)},N^{(2)}) &=S^{(1)}(U^{(1)},V^{(1)},N^{(1)})+S^{(2)}(U^{(2)},V^{(2)},N^{(2)})\ ({\because}3.13) &\\ &=K(U^{(1)}V^{(1)}N^{(1)})^{1/3}+K(U^{(2)}V^{(2)}N^{(2)})^{1/3}\ ({\because}3.5) &\\ &=K[(U^{(1)}V^{(1)}N^{(1)})^{1/3} + (U^{(2)}V^{(2)}N^{(2)})^{1/3}] & \end{flalign*} \]

\(U^{(1)},V^{(1)},N^{(1)},U^{(2)},V^{(2)},N^{(2)}\) は独立変数とする。\(K\) は定数とする

\(C_0:\ U=U^{(1)}+U_1^{(2)},V=V^{(1)}+V_1^{(2)},N=N^{(1)}+N_1^{(2)}\)  (4.3) とする

\(U,V,N\) は独立変数とする

\[ \begin{flalign*} \hat{S_1}(U,V,N,U^{(1)},V^{(1)},N^{(1)}) &=\hat{S}(U^{(1)},V^{(1)},N^{(1)},U-U^{(1)},V-V^{(1)},N-N^{(1)}) &\\ &= K[(U^{(1)}V^{(1)}N^{(1)})^{1/3} +((U-U^{(1)})(V-V^{(1)})(N-N^{(1)}))^{1/3}] & \end{flalign*} \] とする

\(C_1:\ N_1^{(1)}=\displaystyle\frac{N}{2}\) とする

\[ \begin{flalign*} \displaystyle\hat{S_2}(U,V,N,U^{(1)},V^{(1)}) &=\hat{S_1}(U,V,N,U^{(1)},V^{(1)},N_1^{(1)}) &\\ &=K\Big(\frac{N}{2}\Big)^{1/3} [ (U^{(1)}V^{(1)})^{1/3} +((U-U^{(1)})(V-V^{(1)}))^{1/3}]\ \ (4.7) & \end{flalign*} \] とする

\(C_2:\ 0\le V_1^{(1)}\le V\) とする

\[ \begin{flalign*} \displaystyle\hat{S_3}(U,V,N,U^{(1)}) &=\hat{S}_2(U,V,N,U^{(1)},V_1^{(1)}) &\\ &= K\Big(\frac{N}{2}\Big)^{1/3} [(U^{(1)}V_1^{(1)})^{1/3} +((U-U^{(1)})(V-V_1^{(1)}))^{1/3}] & \end{flalign*} \] とする

\[ \begin{flalign*} S(U,V,N:C_0,C_1,C_2) &=\max_{U^{(1)},V^{(1)},N^{(1)},U^{(2)},V^{(2)},N^{(2)}}(\hat{S},C_0,C_1,C_2)\ ({\because}\text{要請II(v)}) &\\ &=\max_{U^{(1)},V^{(1)},N^{(1)}}(\hat{S_1},C_1,C_2)\ ({\because}C_0) &\\ &=\max_{U^{(1)},V^{(1)}}(\hat{S_2},C_2)\ ({\because}C_1) &\\ &=\max_{U^{(1)}}\hat{S_3}\ ({\because}C_2) & \end{flalign*} \]

任意の\(U,V,N\)に対して上の等式が成立する。よって \(\displaystyle\max_{U^{(1)}}\hat{S_3}\) が存在すれば \(S(U,V,N:C_0,C_1,C_2)\) は存在する

\(\displaystyle\max_{U^{(1)}}\hat{S_3}\) を求める

\(0\le U^{(1)}\le U\) で \(\hat{S_3}\)は連続である  \(\textcolor{red}{(*1)}\)

\[ \begin{flalign*} &&\textcolor{red}{(*1)}&(U^{(1)})^{1/3}は連続({\because}\ p>0 ならば x^p は連続)\\ &&& (U-U^{(1)})^{1/3}は連続({\because}\ 連続関数の合成関数は連続)\\ &&& よって \hat{S_3} は連続({\because}\ 連続関数の線形性) \end{flalign*} \]

\(0<U^{(1)}<U\) とする

\[ \begin{flalign*} \frac{\partial\hat{S}_3\ }{\partial U^{(1)}} &=K\Big(\frac{N}{2}\Big)^{1/3} \Big[\frac{1}{3}(U^{(1)})^{-2/3}(V_1^{(1)})^{1/3} +\frac{1}{3}(-1)(U-U^{(1)})^{-2/3}(V-V_1^{(1)})^{1/3}\Big]\ \textcolor{red}{(*2)} &\\ &=K\Big(\frac{N}{2}\Big)^{1/3} \frac{1}{3} \Big[(U^{(1)})^{-2/3}(V_1^{(1)})^{1/3} -(U-U^{(1)})^{-2/3}(V-V_1^{(1)})^{1/3}\Big] & \end{flalign*} \]

\[ \begin{flalign*} &&\textcolor{red}{(*2)}&0<a<1,\ x>0 ならば(x^a)'=ax^{a-1}\\ \end{flalign*} \]

\(0<u_1<u_2<U\) とする

\[ \begin{flalign*} \frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=u_2} -\frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=u_1} &=K\Big(\frac{N}{2}\Big)^{1/3} \frac{1}{3} \Big[(u_2^{-2/3}-u_1^{-2/3})(V_1^{(1)})^{1/3} -((U-u_2)^{-2/3}-(U-u_1)^{-2/3})(V-V_1^{(1)})^{1/3}\Big] & \end{flalign*} \]

\(0<u_1<u_2\) なので \(u_1^{-2/3}>u_2^{-2/3}\ ({\because}\ 0<a<b,p<0 ならば a^p>b^p)\)

\({\therefore}u_2^{-2/3}-u_1^{-2/3}<0\)

また \(-u_1>-u_2\) なので

\({\therefore}U-u_1>U-u_2>0\)

\({\therefore}(U-u_1)^{-2/3}<(U-u_2)^{-2/3}\)

\({\therefore}(U-u_2)^{-2/3}-(U-u_1)^{-2/3}>0\)

\({\therefore}\displaystyle\frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=u_2} -\frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=u_1} < 0\)

よって \(\displaystyle\frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\) は強単調減少である

よって \(\displaystyle\frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=U_1^{(1)}}=0,\ 0<U_1^{(1)}<U\) ならば \(U_1^{(1)}\) で \(\hat{S_3}\) は最大となる  \(\textcolor{red}{(*3)}\)

\[ \begin{flalign*} &&\textcolor{red}{(*3)}& a\le x\le b で f(x)が連続 \\ &&&a<x<bでf'(x)が強単調減少とする \\ &&&a<x_0<bでf'(x_0)=0とする\\ &&&a\le x<x_0 のとき\\ &&&f'(c)=\frac{f(x_0)-f(x)}{x_0-x},\ x<c<x_0({\because}\ 平均値の定理) \\ &&&f'(c)>f'(x_0)=0({\because}\ f'(x)は強単調減少) \\ &&&{\therefore}\frac{f(x_0)-f(x)}{x_0-x}>0 \\ &&&x_0-x>0 なので\\ &&&{\therefore}f(x_0)-f(x)>0 \\ &&&{\therefore}f(x_0)>f(x) \\ &&&x_0<x\le b のとき\\ &&&f'(c)=\frac{f(x)-f(x_b)}{x-x_0},\ x_0<c<x({\because}\ 平均値の定理) \\ &&&f'(c)<f'(x_0)=0 なので \\ &&&\frac{f(x)-f(x_0)}{x-x_0}<0\\ &&&x-x_0>0 なので \\ &&&f(x)-f(x_0)<0\\ &&&{\therefore}f(x)<f(x_0)\\ &&&よって a\le x\le b,\ x\ne x_0 ならば f(x) < f(x_0) \\ &&&よって x_0 で f(x) は最大となる \end{flalign*} \]

\(\displaystyle\frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=U_1^{(1)}}=0\) , \(0<U_1^{(1)}<U\) とする

\((U_1^{(1)})^{-2/3}(V_1^{(1)})^{1/3}-(U-U_1^{(1)})^{-2/3}(V-V_1^{(1)})^{1/3}=0\)

\({\therefore}(U_1^{(1)})^{-2/3}(V_1^{(1)})^{1/3}=(U-U_1^{(1)})^{-2/3}(V-V_1^{(1)})^{1/3}\)

\(V_1^{(1)}\ne 0\) とする

\({\therefore}\displaystyle\Big(\frac{U_1^{(1)}}{U-U_1^{(1)}}\Big)^{-2/3} =\Big(\frac{V-V_1^{(1)}}{V_1^{(1)}}\Big)^{1/3}\)

\({\therefore}\displaystyle\Big(\frac{U-U_1^{(1)}}{U_1^{(1)}}\Big)^2 =\frac{V-V_1^{(1)}}{V_1^{(1)}}\)

\({\therefore}\displaystyle\frac{U-U_1^{(1)}}{U_1^{(1)}} =\pm\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}\)

\(\displaystyle\frac{U-U_1^{(1)}}{U_1^{(1)}} =\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}\) とすると

\(\displaystyle U_1^{(1)}=\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\) となる

逆に \(V_1^{(1)}\ne 0,\ V_1^{(1)}\ne V\) かつ \(\displaystyle U_1^{(1)}=\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\) とすると

\(U>0かつ\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}>0\) なので \(0<U_1^{(1)}<U\) である

また

\[ \begin{flalign*} \frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=U_1^{(1)}} &=K\Big(\frac{N}{2}\Big)^{1/3} \frac{1}{3} \Big[(U_1^{(1)})^{-2/3}(V_1^{(1)})^{1/3} -(U-U_1^{(1)})^{-2/3}(V-V_1^{(1)})^{1/3}\Big] &\\ &=K\Big(\frac{N}{2}\Big)^{1/3}\frac{1}{3} \Bigg[\Bigg(\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\Bigg)^{-2/3}(V_1^{(1)})^{1/3} -\Bigg(U-\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\Bigg)^{-2/3}(V-V_1^{(1)})^{1/3}\Bigg] &\\ &=K\Big(\frac{N}{2}\Big)^{1/3}\frac{1}{3} \Bigg[\Bigg(\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}(V^{(1)})^{-1/2}\Bigg)^{-2/3} -\Bigg(\frac{U\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}(V-V_1^{(1)})^{-1/2}\Bigg)^{-2/3}\Bigg] &\\ &=K\Big(\frac{N}{2}\Big)^{1/3}\frac{1}{3} \Bigg[\Bigg(\frac{U(V^{(1)})^{-1/2}}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\Bigg)^{-2/3} -\Bigg(\frac{U(V^{(1)})^{-1/2}}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\Bigg)^{-2/3}\Bigg] &\\ &=0& \end{flalign*} \] である

よって \(V_1^{(1)}\ne 0,\ V_1^{(1)}\ne V\) ならば

\(\displaystyle U_1^{(1)}=\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\) (4.11) において \(\hat{S_3}\) は最大となる

よって

\(0<V_1^{(1)}<V\) ならば

\[ \begin{flalign*} S(U,V,N:C_0,C_1,C_2) &=\max_{U^{(1)}}\hat{S_3} =\frac{\partial\hat{S}_3\ }{\partial U^{(1)}}\Big|_{U^{(1)}=U_1^{(1)}} &\\ &=K\Big(\frac{N}{2}\Big)^{1/3} \Bigg[\Bigg(\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}V_1^{(1)}\Bigg)^{1/3} + \Bigg(\Bigg(U-\frac{U}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\Bigg)(V-V_1^{(1)})\Bigg)^{1/3}\Bigg] &\\ &=K\Big(\frac{UN}{2}\Big)^{1/3} \Bigg[\Bigg(\frac{V_1^{(1)}}{1+\sqrt{\frac{V-V_1^{(1)}}{V_1^{(1)}}}}\Bigg)^{1/3} +\Bigg(\frac{V-V_1^{(1)}}{1+\sqrt{\frac{V_1{(1)}}{V-V_1^{(1)}}}}\Bigg)^{1/3}\Bigg]\ \ (4.10) & \end{flalign*} \]

\(V_1^{(1)}= 0\) ならば

\(\displaystyle\hat{S_3}=K\Big(\frac{N}{2}\Big)^{1/3}(U-U^{(1)})^{1/3}V^{1/3}\)

この\(\hat{S_3}\) は \(0\le U^{(1)}\le U\) で強単調減少である  \(\textcolor{red}{(*4)}\)

\[ \begin{flalign*} &&\textcolor{red}{(*4)}& 0\le x \le X で (X-x)^{1/3} は強単調減少 \\ &&&(証明) \\ &&&0\le x_1 < x_2 \le X とする\\ &&&{\therefore}X-x_1>X-x_2\ge 0\\ &&&{\therefore}(X-x_1)^{1/3}>(X-x_2)^{1/3}({\because}省略)\\ &&&よって(X-x)^{1/3} は強単調減少 \end{flalign*} \]

よって \(U_1^{(1)}=0\) で \(\displaystyle\hat{S_3}=K\Big(\frac{N}{2}\Big)^{1/3}U^{1/3}V^{1/3}\) は最大となる

\(V_1^{(1)}=V\) ならば

\(\displaystyle\hat{S_3}=K\Big(\frac{N}{2}\Big)^{1/3}(U^{(1)})^{1/3}V^{1/3}\)

この\(\hat{S_3}\) は \(0\le U^{(1)}\le U\) で強単調増加である  \(\textcolor{red}{(*5)}\)

\[ \begin{flalign*} &&\textcolor{red}{(*5)}& 0\le x \le X で x^{1/3} は強単調増加 \\ &&&(証明) 省略 \end{flalign*} \]

よって \(U_1^{(1)}=U\) で \(\displaystyle\hat{S_3}=K\Big(\frac{N}{2}\Big)^{1/3}U^{1/3}V^{1/3}\) は最大となる

よって

\(V_1^{(1)}=V\) または \(V_1^{(1)}=V\) ならば

\(\displaystyle S(U,V,N:C_0,C_1,C_2) =\max_{U^{(1)}}\hat{S_3}=K\Big(\frac{N}{2}\Big)^{1/3}U^{1/3}V^{1/3}\)

本文では \(V_1^{(1)}=0\) または \(V_1^{(1)}=V\) の場合について書かれていないが、 暗黙で \(0<V_1^{(1)}<V\) と仮定されているのかもしれない

(注意)複合系のエントロピー\(S\)は\(U,V,N\)を変化させたときの\(\hat{S}\)の最大値ではない。 \(U^{(1)},V^{(1)},N^{(1)},U^{(2)},V^{(2)},N^{(2)}\)を変化させたときの\(\hat{S}\)の最大値である(要請II(v))
